
model User {
  id    Int    @id @default(autoincrement())
  name  String
  email String @unique
}

datasource db {
  provider = "postgresql" // Ou "mysql" ou "sqlite"
  url      = env("DATABASE_URL")
}

  "scripts": {

    "start": "concurrently \"npm run start:backend\" \"npm run start:frontend\"",
  }

npm run start:backend
